```bash
npm install --production

# 安装PM2并启动
npm install -g pm2
pm2 start server.js --name bilibili-parser
pm2 startup && pm2 save
```

### 5. 配置防火墙
```bash
# CentOS/RHEL 开放端口
firewall-cmd --zone=public --add-port=3000/tcp --permanent
firewall-cmd --reload

# Ubuntu/Debian 开放端口
ufw allow 3000/tcp
ufw reload
```

### 6. 配置Nginx反向代理
```bash
# 创建Nginx配置文件
cat > /etc/nginx/sites-available/bilibili-parser.conf << 'EOF'
server {
    listen 80;
    server_name your-domain.com;

    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_read_timeout 86400;
    }
}
EOF

# 创建软链接
ln -s /etc/nginx/sites-available/bilibili-parser.conf /etc/nginx/sites-enabled/

# 测试配置并重启
nginx -t
systemctl restart nginx
```

## 七、常见问题及解决

### 1. 502 Bad Gateway 错误
**解决方法：**
```bash
pm2 status
netstat -tulpn | grep 3000
firewall-cmd --list-ports
```

### 2. 403 Forbidden 错误
**解决方法：**
```bash
chown -R www:www /www/wwwroot/bilibili-parser
chmod -R 755 /www/wwwroot/bilibili-parser
```

### 3. 服务自动停止
**解决方法：**
```bash
pm2 startup
pm2 save
```
